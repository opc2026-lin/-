/**

* 核心清算对比函数

*/

function calculateFujianTariff() \{

const db = FUJIAN\_ELECTRICITY\_DB[1]; // 精准清算1月份

// 💡 跨期尖峰电量非7-9月安全归流并入高峰

let q\_sharp = USER\_META.q\_sharp;

let q\_peak = USER\_META.q\_peak;

let is\_sharp\_merged = false;

if (q\_sharp > 0) \{

q\_peak = USER\_META.q\_peak + USER\_META.q\_sharp; // 102628 + 34634 = 137262 kWh

q\_sharp = 0;

is\_sharp\_merged = true;

}

const q\_total = q\_sharp + q\_peak + USER\_META.q\_flat + USER\_META.q\_valley; // 256893 kWh

// ----------------------------------------------------------------

// 路径 A 真实生存现状：用户当前正在执行的【老零售交易合同成本】

// ----------------------------------------------------------------

const current\_contract\_energy\_fee = (q\_peak * USER\_META.current\_contract\_p\_peak) +

(USER\_META.q\_flat * USER\_META.current\_contract\_p\_flat) +

(USER\_META.q\_valley * USER\_META.current\_contract\_p\_valley);

const current\_contract\_total\_cost = current\_contract\_energy\_fee + USER\_META.actual\_addons\_total\_fee;

const current\_contract\_avg\_price = current\_contract\_total\_cost / q\_total;

// ----------------------------------------------------------------

// 路径 B 推荐优化方案一：切换为福建亿云能源【全新零售固定价格套餐】

// 假设我们给他批复的平时段基准价 p\_fixed\_base 为 0.380 元/kWh

// ----------------------------------------------------------------

const p\_fixed\_base = 0.380;

const p\_fixed\_peak = p\_fixed\_base * 1.58;

const p\_fixed\_flat = p\_fixed\_base * 1.0;

const p\_fixed\_valley = p\_fixed\_base * 0.37;

const fixed\_energy\_fee = (q\_peak * p\_fixed\_peak) + (USER\_META.q\_flat * p\_fixed\_flat) + (USER\_META.q\_valley * p\_fixed\_valley);

const fixed\_total\_cost = fixed\_energy\_fee + USER\_META.actual\_addons\_total\_fee;

const fixed\_avg\_price = fixed\_total\_cost / q\_total;

// ----------------------------------------------------------------

// 路径 C 推荐优化方案二：切换为福建亿云能源【全新零售分时均价联动套餐】

// 全电量 * 采购均价 * 1.05

// ----------------------------------------------------------------

const sharing\_energy\_fee = q\_total * (db.sales\_purchase\_avg * 1.05);

const sharing\_total\_cost = sharing\_energy\_fee + USER\_META.actual\_addons\_total\_fee;

const sharing\_avg\_price = sharing\_total\_cost / q\_total;

// 计算动态对比直方图的高度映射百分比 (以三者单价最高者作为 100% 满格)

const max\_price = Math.max(current\_contract\_avg\_price, fixed\_avg\_price, sharing\_avg\_price);

const height\_current = ((current\_contract\_avg\_price / max\_price) * 100).toFixed(1);

const height\_fixed = ((fixed\_avg\_price / max\_price) * 100).toFixed(1);

const height\_sharing = ((sharing\_avg\_price / max\_price) * 100).toFixed(1);

// 套cam寻优决策

let best\_package = "";

let potential\_saving = 0;

if (sharing\_total\_cost \< fixed\_total\_cost) \{

best\_package = "分时均价联动套餐";

potential\_saving = fixed\_total\_cost - sharing\_total\_cost;

} else \{

best\_package = "零售固定价格套餐";

potential\_saving = sharing\_total\_cost - fixed\_total\_cost;

}

const total\_saved\_vs\_current = current\_contract\_total\_cost - Math.min(fixed\_total\_cost, sharing\_total\_cost);

return \{

q\_total, q\_peak, q\_flat: USER\_META.q\_flat, q\_valley: USER\_META.q\_valley, raw\_q\_sharp: USER\_META.q\_sharp, is\_sharp\_merged,

current\_contract\_energy\_fee, current\_contract\_total\_cost, current\_contract\_avg\_price,

fixed\_energy\_fee, fixed\_total\_cost, fixed\_avg\_price,

sharing\_energy\_fee, sharing\_total\_cost, sharing\_avg\_price,

height\_current, height\_fixed, height\_sharing,

best\_package, potential\_saving, total\_saved\_vs\_current

};

}
