# 1. 意图识别：判断是否调用 RAG
def _intent_recognize(self, query):
    prompt = f"分析需求：{query}，只需回答“是”或“否”"
    return self.llm.predict(prompt).strip()

# 2. 智能体核心决策逻辑
def agent_decide(self, input_data):
    need_rag = self._intent_recognize(input_data["query"])

# 3. 动态分支：按需调用工具
if need_rag == "是":
    ctx = self.tools["rag_search"](input_data["query"])
    self.memory.append({"used_rag": need_rag})

# 4. 记忆更新
input_data["retrieved_ctx"] = ctx or ""
return input_data

def workflow_router(input_data):

while current_node and state["retry_count"] <= 2:

    # 根据审核结果动态决定下一节点
    if "审核结论：通过" in input_data["check_result"]:

       return "node_generate_plan" # 重试节点

    if current_node == "node_ai_check":
       # 审核不通过时触发重试计数+1
       current_node = workflow_router(state)

    if current_node == "node_generate_plan":
        state["retry_count"] += 1

 核心逻辑解析：闭环控制与容错机制

• 状态机驱动：使用 `while` 循环作为主引擎，通过 `current_node` 变量驱动不同阶段的任务节点（生成、审核、归档）。

• 有限重试：设置 `max_retries=2` 阈值，审核失败时自动重试并累加计数，防止陷入死循环，超限后终止并告警。
